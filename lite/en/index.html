There is a solve of codewars kata Data Reverse.

Time complexity O(n) because in this loop it swaps only n/2 items. Space complexity O(1), so it uses only one space for swap. No array methods, no any hidden time or space complexity.

const dataReverse = data => {
  let left, right, i, t;
  for (left = 0, right = data.length - 8; left < right; left += 8, right -= 8)
    for (i = 0; i < 8; ++i) {
      t = data[left + i]; 
      data[left + i] = data[right + i];
      data[right + i] = t;
    }
  return data;
};
